165:     public function move($dest)
166:     {
167:         @mkdir(dirname($dest), 0777, TRUE); // @ - dir may already exist
168:         if (!call_user_func(is_uploaded_file($this->tmpName) ? 'move_uploaded_file' : 'rename', $this->tmpName, $dest)) {
